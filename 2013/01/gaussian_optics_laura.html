In [318]:
 
# Rayleigh range function (w_0 - beam waist)
def z_r(w_0, wavelength): 
    return (pi*w_0**2)/wavelength
 
# Beam radius function (z - distance from waist, w_0 - beam waist, z_r - Rayleigh range)
def radius(z, w_0, z_r): 
    return w_0*(1+(z/z_r)**2)**0.5
 
# Gaussian transfer matrix function (q - beam complex q-parameter, A,B,C,D elements of transformation matrix)
def gaussian_transfer(q, A, B, C, D): 
    return (A*q+B)/(C*q+D)

Assuming this is the same beam size as the one at Laura's trap, we calculate the complex q parameter qi and apply the transformation associated with Laura's focussing lens (f=250mm) i.e. qf=Aqi+BCqi+D, with A=1, B=0, C=−1/f, D=1.

In [340]:
 
z_c = z_r(w_c, wavelength) # Rayleigh range of `collimated' beam
q_i = complex(0,z_c) #complex q parameter of collimated beam
f = 250e-3 #Focal length of Laura's lens
q_f = gaussian_transfer(q_i,1,0,-1/f,1) # thin lens transformation, focal length f 
print 'q_f.real = {0:.0f} mm, (distance to focus). \nq_f.imag = {1:.0f} mm, (Rayleigh range).\nDiscrepancy of q_f.real with focal length (f = 250 mm) comes from deviation from perfectly collimated beam (z_c = inf).'.format(q_f.real*1e3, q_f.imag*1e3)
